Stack

Stack is LIFO (Last In First Out) data structures. In LIFO data structure element inserted (or appended) at last is removed first. We are goind to implement stack using linked list. 

            
        class _Node:
            def __init__(self, data=None, pre=None) -> None:
                self._data = data 
                self._pre = pre

        class Stack:
            def __init__(self) -> None:
                self._tail = _Node()
                self._size = 0

            def __len__(self):
                return self._size 

            def is_empty(self):
                return self._size == 0

            def append(self, data):
                newNode = _Node(data, self._tail)
                self._tail = newNode
                self._size += 1

            def pop(self):
                removeItem = self._tail
                self._tail = self._tail._pre
                self._size -= 1
                return removeItem._data

            def print(self):
                __copy = self._tail
                __size = self._size
                print("{:>10} {:<10}".format("Index", "Data"))
                while __copy._pre is not None:
                    print("{:>10} {:<10}".format(__size, __copy._data))
                    __copy = __copy._pre
                    __size -= 1
                
                del __copy
                del __size

        def main():
            stk = Stack()
            for i in range(10):
                stk.append(i)
            stk.print()
            print(stk.pop())
            stk.print()

        if __name__ == "__main__":
            main()
